OBR.contextMenu.create({ id: "com.simple.marker/mark", items: [ { id: "mark-token", label: "Mark Token", icon: "https://www.owlbear.rodeo/logo.svg", filter: { every: [{ key: "type", value: "IMAGE" }] }, async onClick(context) { const tokenIds = context.items.map(item => item.id); const existingAttachments = await OBR.scene.items.getItemAttachments(tokenIds); const markerIdsToRemove = existingAttachments .filter(item => item.metadata["com.simple.marker/isMarker"]) .map(item => item.id); if (markerIdsToRemove.length > 0) { await OBR.scene.items.deleteItems(markerIdsToRemove); } else { const markers = context.items.map((token) => { return OBR.scene.items.buildShape() .width(token.width) .height(token.height) .shapeType("CIRCLE") .fillColor("#ff0000") .fillOpacity(0.4) .strokeColor("#000000") .strokeWidth(2) .position(token.position) .attachedTo(token.id) .layer("ATTACHMENT") .metadata({ "com.simple.marker/isMarker": true }) .disableHit(true) .build(); }); await OBR.scene.items.addItems(markers); } } } ] });
